def add_info(df:pd.DataFrame, name):
    df["Submit"] = pd.to_datetime(df["Submit"], errors="coerce")
    df.index = df["Submit"]
    df = df[df["NodeHours"]>0]
    df =df.rename(columns={
        "NodeHours":f"{name} usage",
        })
    return df
